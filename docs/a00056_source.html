59 //disabling the interrupt
60 IEC4bits.CMP3IE = 0;

101void __attribute__ ( ( __interrupt__ , auto_psv ) ) _ADCAN0Interrupt ( void )
102{
103 GPIO_debug_SetHigh(); // to keep track of the interrupt latency per cycle
104 //Read all the ADC value from the ADCBUF
105 psfb_ptr->Data.ISensePrimary = ADCBUF0;
106 psfb_ptr->Data.ISenseSecondary = ADCBUF1;
107
108 psfb_ptr->ISecAveraging.Accumulator += psfb_ptr->Data.ISenseSecondary;
109 psfb_ptr->ISecAveraging.Counter = psfb_ptr->ISecAveraging.Counter + 1;
110
111 PwrCtrl_UpdateADConverterData();
112
113 ADC1_SoftwareTriggerEnable();
114
115 // Execute the fault detection
116 Fault_Execute(); // moving it here, so that all other faults are checked other than VCAP undervoltage
117
118 if (psfb_ptr->State > 1) { // if state is greater than precharge, give the phase shift control over to controller
119 ControlLoop_Interrupt_CallBack(); //update software based ADC, execute Faults
120 }
121
122 PwrCtrl_PWM_UpdateSecondaryRectifiers(); // check Ishunt within range
123
124 GPIO_debug_SetLow();
125 //clear the FB_P_CT_FILT interrupt flag
126 IFS5bits.ADCAN0IF = 0;
127}
