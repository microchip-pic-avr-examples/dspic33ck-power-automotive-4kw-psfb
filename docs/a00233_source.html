193void App_PBV_psfb_Build_Frame()
194{
195 // send back one "flag word" which combines fault and status and enable control flag
196 uint16_t enabled = Dev_PwrCtrl_Get_EnableFlag();
197 uint16_t fault_flags = Fault_GetFlags();
198 uint16_t status_flags = Dev_PwrCtrl_Get_Status();
199 uint16_t flag_word = enabled + ((status_flags & 0x0003)<<1) + (fault_flags<<3);
200
201 buffer_sixteen_tx[0] = (1<<Dev_PwrCtrl_Get_State());
202 buffer_sixteen_tx[1] = flag_word;
203 buffer_sixteen_tx[2] = FAULT_EN_GetValue() +
204 (psfb_ptr->SecRec.SR_Flag<<1) + (psfb_ptr->ILoop.Enable<<2) + (psfb_ptr->ILoop.Enable<<3)+ (psfb_ptr->Precharge.PrechargeEnabled<<4);
205 buffer_sixteen_tx[3] = Dev_PwrCtrl_GetAdc_Vpri() ;
206 buffer_sixteen_tx[4] = PwrCtrl_GetAdc_Vrail_5V();
207 buffer_sixteen_tx[5] = (PwrCtrl_GetAdc_Temperature() > 497) ? (PwrCtrl_GetAdc_Temperature() - 497 ) : 0;
208
209 // I-CT
210 buffer_sixteen_tx[6] = PwrCtrl_GetAdc_Ipri_ct();
211 buffer_sixteen_tx[7] = DAC3DATH;
212 buffer_sixteen_tx[8] = Dev_PwrCtrl_GetIPri_Offset();
213 buffer_sixteen_tx[9] = DAC3DATH - Dev_PwrCtrl_GetIPri_Offset();
214
215 // I-SHUNT
216 buffer_sixteen_tx[10] = PwrCtrl_GetAdc_Isec_shunt();
217 buffer_sixteen_tx[11] = Dev_PwrCtrl_GetIsec_Offset();
218 //buffer_sixteen_tx[12] = psfb_ptr->Properties.IReference;
219 buffer_sixteen_tx[12] = psfb_ptr->VLoop.Output; // output of vloop will be reference of the iloop
220 buffer_sixteen_tx[13] = psfb_ptr->ILoop.Output;
221 buffer_sixteen_tx[14] = psfb_ptr->SecRec.Threshold_high;
222 buffer_sixteen_tx[15] = psfb_ptr->SecRec.Threshold_low;
223
224 // Vout
225 buffer_sixteen_tx[16] = Dev_PwrCtrl_GetAdc_Vsec();
226
227 // Vcap
228 buffer_sixteen_tx[17] = Dev_PwrCtrl_GetVoltage_Vcap();
229 buffer_sixteen_tx[18] = psfb_ptr->VLoop.Reference;
230 buffer_sixteen_tx[19] = psfb_ptr->Properties.VSecReference;
231 buffer_sixteen_tx[20] = psfb_ptr->VLoop.Output;
232
233 // phase
234 buffer_sixteen_tx[21] = PG1TRIGC;
235 buffer_sixteen_tx[22] = psfb_ptr->Precharge.maxDutyCycle;
236
237 //(PwrCtrl_GetAdc_Temperature() > 497) ? (PwrCtrl_GetAdc_Temperature() - 497 ) : 0;
238 //temp_ = psfb_ptr->ISecAveraging.AverageValue;
239 buffer_sixteen_tx[23] = psfb_ptr->ISecAveraging.AverageValue;
240 buffer_sixteen_tx[24] = psfb_ptr->Droop.ref_diff;
241
242 buffer_sixteen_tx[25] = psfb_ptr->VoutCalibrate.calculated_value;
243 buffer_sixteen_tx[26] = psfb_ptr->VoutCalibrate.real_value;
244 buffer_sixteen_tx[27] = (uint16_t)psfb_ptr->VoutCalibrate.gain_factor;
245
246 PBV_Change_from_Sixteen_to_Eight(buffer_sixteen_tx, buffer_eight_tx, 28);
247
248 App_PBV_psfb_TX_Ptr->Data_Buffer = buffer_eight_tx;
249 App_PBV_psfb_TX_Ptr->Length = 28 * 2 ;
250}

260void App_PBV_psfb_Process_Buttons(uint16_t * data) {
261 uint16_t switchcase = data[0];
262 switch (switchcase) {
263 case 0x0001:
264 if (psfb_ptr->Precharge.precharged == 1)
265 PwrCtrl_SetEnable(true);
266 break;
267 case 0x0000:
268 PwrCtrl_SetEnable(false);
269 psfb_ptr->Precharge.PrechargeEnabled = 0;
270 //FAULT_EN_SetLow();
271 break;
272
273 case 0xaa00:
274// Dev_Reset_Average_Buffer();
275 Fault_Reset();
276 PwrCtrl_Reset();
277 break;
278
279 case 0x00aa:
280 //FAULT_EN_SetHigh();
281 // charge_en = 1;
282
283 PwrCtrl_MaxDutyCycle();
284 psfb_ptr->Precharge.PrechargeEnabled = 1;
285 PwrCtrl_PWM_Enable();
286 break;
287 case 0x00bb:
288 FAULT_EN_SetLow();
289 // charge_en = 0;
290 break;
291
292 default:
293 break;
294 }
295// status_controller_state = button_start | (button_start_sync << 1) | (button_random_action << 2)| (dev_fan_data_ptr->override_flag << 3) ;
296}

305void App_PBV_psfb_Process_Sliders(uint16_t * data) {
306 uint16_t switchcase = data[0];
307 uint16_t temp = 0;
308
309 Nop();
310 Nop();
311 Nop();
312 Nop();
313
314 switch (switchcase) {
315 case 0xaa:
316 if (psfb_ptr->ILoop.Enable == 1)
317 break;
318 else {
319 slider_PS_PP = data[1];
320
321 PwrCtrl_SetPhaseTarget(data[1]);
322 //PWM_TriggerCCompareValueSet(1, data[1] * 56);
323 //GPIO_Debug_high();
324 PwrCtrl_PWM_Update();
325 //GPIO_Debug_Low();
326 // PG1STATbits.UPDREQ = 1; // Set manual update (can be automated later)
327 }
328 break;
329
330 case 0xbb:
331 //current slider ref -- DACDATH update here.
332 dead_time_right = data[1];
333// temp = 0.5 + (data[1] * 0.18);
334// temp = ( temp * 4095 )/3.3 ;
335// current_dac_value = temp;
336// DAC1DATH = temp;
337 PG1DTH = data[1]; // set dead time to 150ns
338 PG1DTL = data[1]; // set dead time to 150ns
339
340 PG3DTH = data[1]; // set dead time to 150ns
341 PG3DTL = data[1]; // set dead time to 150ns
342// PG2DTH = 0; // no dead time on PWMxH
343// PG2DTL = 1000; // set dead time to 250ns
344// PG3DTH = data[1]; // set dead time to 150ns
345// PG3DTL = data[1]; // set dead time to 150ns
346// PG4DTH = 0; // no dead time on PWMxH
347// PG4DTL = 1000; // set dead time to 250ns
348 PG1STATbits.UPDREQ = 1; // Set manual update (can be automated later)
349
350 break;
351
352 case 0xcc:
353 //voltage_ref = data[1];
354 //PwrCtrl_SetVSecReference(voltage_ref);
355 current_ref = data[1];
356 PwrCtrl_SetIReference(current_ref);
357 break;
358
359 case 0xee:
360 voltage_ref = data[1];
361 float vreftemp = (float)voltage_ref * psfb_ptr->VoutCalibrate.gain_factor;
362 if (abs(vreftemp - voltage_ref)>372) vreftemp = (float)voltage_ref; // if difference is greater than 0.3 v
363 PwrCtrl_SetVSecReference((uint16_t)vreftemp);
364 psfb_ptr->Droop.Droop_Voltage_Reference_from_PBV = (uint16_t)vreftemp;
365 break;
366
367 case 0xdd:
368 //current slider ref -- DACDATH update here.
369// temp = data[1];
375// current_slider_ref = temp;
376 DAC3DATH = data[1];
377 break;
378
379 default:
380 break;
381 }

391void App_PBV_psfb_Frame_Parser(uint16_t protocol_ID, uint16_t length, uint8_t * data){
392 PBV_Change_from_Eight_to_Sixteen(data, buffer_sixteen_rx, length);

396 break;
397 case PBV_GUI_SLIDERS:
398 App_PBV_psfb_Process_Sliders(buffer_sixteen_rx);
