178inline static uint16_t ADC1_ConversionResultGet( enum ADC_CHANNEL channel )
179{
180 uint16_t result = 0x0U;
181
182 switch(channel)
183 {
184 case FB_VOUT:
185 result = ADCBUF2;
186 break;
187 case FB_VCAP:
188 result = ADCBUF9;
189 break;
190 case VIN_INT_AN:
191 result = ADCBUF10;
192 break;
193 case FB_TEMP:
194 result = ADCBUF14;
195 break;
196 case Channel_AN15:
197 result = ADCBUF15;
198 break;
199 case FB_5V:
200 result = ADCBUF19;
201 break;
202 case FB_P_CT_FILT:
203 result = ADCBUF0;
204 break;
205 case I_SEC_AVG_FILT:
206 result = ADCBUF1;
207 break;
208 default:
209 break;
210 }
211 return result;
212}
