192void PwrCtrl_MaxDutyCycle(void){
193 // 10 volts
194 // for 10 volts
195 // divided by 8 to keep it within 16bits
196 psfb.Precharge.scaler = 10 * 2 * 17 * 54;
197 psfb.Precharge.primaryVoltage = Dev_PwrCtrl_GetAdc_Vpri() - 205;
198 // divided by 8 as scaler is also divided by 8
199 psfb.Precharge.primaryVoltage = psfb.Precharge.primaryVoltage >> 3;
200
201 psfb.Precharge.maxDutyCycle = __builtin_divud(psfb.Precharge.scaler, psfb.Precharge.primaryVoltage);
202
203 if (psfb.Precharge.maxDutyCycle > 80)
204 psfb.Precharge.maxDutyCycle = 80; // limit precharge to 80 percent dutycycle
205}
